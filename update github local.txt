import subprocess

# Change directory
repo_path = r"C:\Program Files\BAD\blackaquapydroid"
command_cd = f'cd "{repo_path}"'

# Fetch changes from the remote repository
command_fetch = 'git fetch origin'

# Merge changes into the local branch
command_merge = 'git merge origin/main'

# Combine commands
combined_command = f'{command_cd} && {command_fetch} && {command_merge}'

# Run the commands
try:
    subprocess.run(combined_command, shell=True, check=True)
    print("Git commands executed successfully.")
except subprocess.CalledProcessError as e:
    print(f"Error executing Git commands: {e}")
